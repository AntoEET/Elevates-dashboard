SIMPLE IMPORT SCRIPT FOR ELEVATES DASHBOARD
==============================================

STEP 1: Open Elevates Dashboard
- Go to: http://localhost:3001/crm

STEP 2: Open Browser Console
- Press F12
- Click "Console" tab

STEP 3: Copy and paste this entire script into the console and press Enter:

```javascript
(async () => {
    try {
        console.log('Fetching prospects from Outreach System...');

        const response = await fetch('http://localhost:5000/api/prospects');
        const outreachProspects = await response.json();

        console.log(`Fetched ${outreachProspects.length} prospects from Outreach System`);

        // Map to Elevates format
        const elevatesProspects = outreachProspects.map(p => {
            // Map status to stage
            let stage = 'new-lead';
            if (p.status === 'Contacted') stage = 'invited';
            if (p.status === 'Replied') stage = 'connected';
            if (p.status === 'Meeting') stage = 'meeting-scheduled';
            if (p.status === 'Qualified') stage = 'proposal-sent';
            if (p.status === 'Won') stage = 'closed-won';
            if (p.status === 'Lost') stage = 'closed-lost';

            // Priority
            let priority = 'medium';
            if (p.reply_type === 'interested' || p.meeting_booked) priority = 'high';

            return {
                id: p.prospect_id || `P${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
                name: `${p.first_name || ''} ${p.last_name || ''}`.trim() || 'Unknown',
                company: p.company || '',
                email: p.email || '',
                phone: p.phone || '',
                linkedinProfile: p.linkedin_url || '',
                twitterProfile: '',
                instagramProfile: '',
                otherProfile: '',

                stage: stage,
                dateAdded: p.date_added || new Date().toISOString(),
                dateInvited: p.email_1_sent || undefined,
                dateConnected: p.reply_date || undefined,
                dateFirstMessage: p.email_1_sent || undefined,
                dateMeetingScheduled: p.meeting_date || undefined,

                accepted: p.replied === 1 || p.replied === true,
                firstEmailDate: p.email_1_sent || undefined,

                followUps: [],
                notes: p.notes || '',
                tags: [p.industry, p.source].filter(Boolean),
                priority: priority,
                source: p.source || 'Outreach System',

                createdAt: p.date_added || new Date().toISOString(),
                updatedAt: p.last_updated || new Date().toISOString()
            };
        });

        console.log(`Mapped ${elevatesProspects.length} prospects to Elevates format`);

        // Get existing prospects
        const existing = localStorage.getItem('elevates-prospects');
        let existingProspects = [];

        if (existing) {
            try {
                const parsed = JSON.parse(existing);
                existingProspects = parsed?.state?.prospects || [];
                console.log(`Found ${existingProspects.length} existing prospects`);
            } catch (e) {
                console.warn('Could not parse existing data');
            }
        }

        // Merge - avoid duplicates by ID
        const existingIds = new Set(existingProspects.map(p => p.id));
        const newProspects = elevatesProspects.filter(p => !existingIds.has(p.id));

        const allProspects = [...existingProspects, ...newProspects];

        console.log(`Adding ${newProspects.length} new prospects`);
        console.log(`Total prospects: ${allProspects.length}`);

        // Save to localStorage
        const storeData = {
            state: {
                prospects: allProspects,
                selectedProspectId: null,
                viewMode: 'pipeline',
                searchQuery: '',
                filterStage: 'all',
                filterPriority: 'all'
            },
            version: 0
        };

        localStorage.setItem('elevates-prospects', JSON.stringify(storeData));

        console.log('‚úÖ Import complete!');
        console.log(`üìä Stats:`);
        console.log(`   - Total: ${allProspects.length}`);
        console.log(`   - New: ${newProspects.length}`);
        console.log(`   - Existing: ${existingProspects.length}`);
        console.log('');
        console.log('üîÑ Refreshing page...');

        // Reload page to see prospects
        setTimeout(() => location.reload(), 1000);

    } catch (error) {
        console.error('‚ùå Import failed:', error);
        console.log('Make sure:');
        console.log('1. Outreach Dashboard is running at http://localhost:5000');
        console.log('2. You are on http://localhost:3001/crm');
    }
})();
```

STEP 4: Wait for the page to refresh

Your prospects should now appear!
